## Usage Example

```rust
use ramp_compliance::kyc::{
    EkycService, EkycVerificationRequest, TenantEkycConfig,
    EkycProviderType, UserProvidedData,
};
use ramp_adapter::{IdDocumentType, LivenessCheckType};

// Create service
let service = EkycService::new_mock();

// Configure for tenant
let config = TenantEkycConfig {
    primary_provider: EkycProviderType::Vnpay,
    fallback_provider: Some(EkycProviderType::FptAi),
    enable_liveness: true,
    liveness_check_type: LivenessCheckType::Active,
    min_face_similarity: 0.75,
    ..Default::default()
};

// Verify user
let request = EkycVerificationRequest {
    tenant_id: "tenant_1".to_string(),
    user_id: "user_1".to_string(),
    id_front_image: front_image_bytes,
    id_back_image: Some(back_image_bytes),
    id_document_type: IdDocumentType::Cccd,
    selfie_image: selfie_bytes,
    liveness_data: Some(video_bytes),
    requested_tier: KycTier::Tier1,
    user_provided_data: UserProvidedData {
        full_name: "NGUYEN VAN A".to_string(),
        id_number: "001234567890".to_string(),
        date_of_birth: "01/01/1990".to_string(),
        address: None,
    },
};

let result = service.verify(request, &config).await?;
```
